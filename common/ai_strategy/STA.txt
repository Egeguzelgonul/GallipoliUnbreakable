consolidate_persia = {
	enable = {
		tag = STA
		OR = {
			country_exists = UZB
			country_exists = PER
			country_exists = AFG
		}
	}
	abort = {
		NOT = {
			country_exists = UZB
			country_exists = PER
			country_exists = AFG
		}
	}
	
	ai_strategy = {
		type = antagonize
		id = "PER"			
		value = 200
	}
	ai_strategy = {
		type = antagonize
		id = "AFG"			
		value = 200
	}	
	ai_strategy = {
		type = antagonize
		id = "UZB"			
		value = 200
	}	
}
